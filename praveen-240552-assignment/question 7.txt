#include <iostream>
using namespace std;
int main() {
	int num;
	cin >> num;    //Reading input from STDIN
	int N=num;
	string a[N];
	for(int i=0;i<N;i++){
		cin>>a[i];
	}
	for(int i=0;i<N;i++){
		int c=0;
		for(int j=0;j<i;j++){
			if(a[j]<a[i]){
				c++;
			}
		}
		cout<<c<<endl;
	}
		// Writing output to STDOUT
}
